## Steps to get started

### 1) Join the waitlist

ZeroGPU is in closed beta. Join the waitlist and get dashboard access once approved.

![Step 1 placeholder](/images/placeholders/step-1.svg)

### 2) Create an API key, project ID, and pick a model

In the dashboard, create an API key and a project ID, then choose a model from the catalog.

![Step 2 placeholder](/images/placeholders/step-2.svg)

### 3) Make your first request

Use the `Responses` API with the summarization model to get a plain-text summary.

![Step 3 placeholder](/images/placeholders/step-3.svg)

## Your first request (summary)

Make a `POST` request to `https://api.zerogpu.ai/v1/responses` with two required headers:

- `x-api-key`: your API key
- `x-project-id`: your project ID

<Tabs>
  <Tab title="cURL">

```bash
curl https://api.zerogpu.ai/v1/responses \
  -H "content-type: application/json" \
  -H "x-api-key: <YOUR_API_KEY>" \
  -H "x-project-id: <YOUR_PROJECT_ID>" \
  -d '{
    "model": "zlm-v1-summary-cloud",
    "input": [{"role":"user","content":"Summarize this article in 3 bullet points."}],
    "text": {"format": {"type": "text"}}
  }'
```

  </Tab>
  <Tab title="Python">

```python
import requests

url = "https://api.zerogpu.ai/v1/responses"
headers = {
    "content-type": "application/json",
    "x-api-key": "<YOUR_API_KEY>",
    "x-project-id": "<YOUR_PROJECT_ID>",
}
payload = {
    "model": "zlm-v1-summary-cloud",
    "input": [{"role": "user", "content": "Summarize this article in 3 bullet points."}],
    "text": {"format": {"type": "text"}},
}

resp = requests.post(url, headers=headers, json=payload)
resp.raise_for_status()
text = resp.json()["output"][0]["content"][0]["text"]
print(text)
```

  <Tab title="Go">

```go
package main

import (
  "bytes"
  "encoding/json"
  "net/http"
)

func main() {
  body := map[string]any{
    "model": "zlm-v1-summary-cloud",
    "input": []map[string]string{{"role": "user", "content": "Summarize this article in 3 bullet points."}},
    "text": map[string]any{"format": map[string]string{"type": "text"}},
  }

  payload, _ := json.Marshal(body)
  req, _ := http.NewRequest("POST", "https://api.zerogpu.ai/v1/responses", bytes.NewBuffer(payload))
  req.Header.Set("content-type", "application/json")
  req.Header.Set("x-api-key", "<YOUR_API_KEY>")
  req.Header.Set("x-project-id", "<YOUR_PROJECT_ID>")

  client := &http.Client{}
  resp, _ := client.Do(req)
  defer resp.Body.Close()

  var out map[string]any
  json.NewDecoder(resp.Body).Decode(&out)
}
```

## Second request (IAB ad signals)

Switch the model to `zlm-v1-iab-classify-cloud` to extract structured ad signals.

<Tabs>
  <Tab title="cURL">

```bash
curl https://api.zerogpu.ai/v1/responses \
  -H "content-type: application/json" \
  -H "x-api-key: <YOUR_API_KEY>" \
  -H "x-project-id: <YOUR_PROJECT_ID>" \
  -d '{
    "model": "zlm-v1-iab-classify-cloud",
    "input": [{"role":"user","content":"Classify this article for ad targeting."}],
    "text": {"format": {"type": "text"}}
  }'
```

  </Tab>
  <Tab title="Python">

```python
import requests
import json

url = "https://api.zerogpu.ai/v1/responses"
headers = {
    "content-type": "application/json",
    "x-api-key": "<YOUR_API_KEY>",
    "x-project-id": "<YOUR_PROJECT_ID>",
}
payload = {
    "model": "zlm-v1-iab-classify-cloud",
    "input": [{"role": "user", "content": "Classify this article for ad targeting."}],
    "text": {"format": {"type": "text"}},
}

resp = requests.post(url, headers=headers, json=payload)
resp.raise_for_status()
raw = resp.json()["output"][0]["content"][0]["text"]
classification = json.loads(raw)
print(classification["content"]["iab_2_2"][:3])
```

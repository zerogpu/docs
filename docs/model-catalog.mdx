The **Model Catalog** lets you browse every available ZeroGPU model and compare **pricing** across the tasks you care about. It's useful when you're selecting which `model` identifier to send to `POST /v1/responses`.

Sections on this page: [At a glance](#at-a-glance) (pricing table), [Detailed model cards](#detailed-model-cards), [Model library by task](#model-library-by-task), and [Using a model from the catalog in an API request](#using-a-model-from-the-catalog-in-an-api-request).

export const byPriority = (list) =>
  list.slice().sort((a, b) => (b.displayPriority || 0) - (a.displayPriority || 0));

export const loadModels = () => {
  if (typeof window === "undefined") return Promise.resolve(SEED_MODELS);
  if (!window.__zgpuModelsPromise) {
    window.__zgpuModelsPromise = fetch(ZGPU_MODELS_URL)
      .then((r) => r.json())
      .then((d) => (d && d.models && d.models.length ? d.models : SEED_MODELS));
  }
  return window.__zgpuModelsPromise;
};

export const modelSlug = (id) => id.toLowerCase().replace(/\./g, "-");
export const taskSlug = (t) => t.toLowerCase().replace(/\s+/g, "-");
export const fmtNum = (n) => (typeof n === "number" ? n.toLocaleString("en-US") : n);
export const fmtPrice = (n) => (typeof n === "number" ? "$" + n.toFixed(2) : "-");
export const cleanText = (s) => (s || "").replace(/\s+/g, " ").trim();
export const cleanUrl = (u) => (u || "").replace(/\s+/g, "");
export const pricingOf = (m) => (m && m.pricing) || {};

{/*
  Rendered as a <span> with a CSS background-image rather than an <img> on
  purpose: Mintlify auto-wraps <img> in zero-width zoom <span>s that collapse
  the logo inside flex layouts. Inline styles also sidestep Tailwind JIT.
*/}
export const ModelLogo = ({ src, alt, size }) => {
  const url = cleanUrl(src);
  if (!url) return null;
  const px = (size || 40) + "px";
  return (
    <span
      role="img"
      aria-label={alt || ""}
      className="inline-block shrink-0 rounded-md border border-zinc-950/10 bg-white dark:border-white/10"
      style={{
        width: px,
        height: px,
        backgroundImage: 'url("' + url + '")',
        backgroundSize: "72%",
        backgroundPosition: "center",
        backgroundRepeat: "no-repeat",
      }}
    />
  );
};

export const Badge = ({ children }) => (
  <span className="rounded-full bg-zinc-950/[0.06] px-2 py-0.5 text-[11px] font-medium text-zinc-600 dark:bg-white/10 dark:text-zinc-300">
    {children}
  </span>
);

export const useModels = () => {
  const [models, setModels] = useState(byPriority(SEED_MODELS));
  useEffect(() => {
    let active = true;
    loadModels()
      .then((m) => {
        if (active && m && m.length) setModels(byPriority(m));
      })
      .catch(() => {});
    return () => {
      active = false;
    };
  }, []);
  return models;
};

{/*
  Rendered as a CSS grid of <div>/<a> rows rather than a <table>: Mintlify
  auto-wraps raw <table> elements in a full-bleed scroll container (negative
  margin + page-padding + per-cell min-width) that clips the left edge. A grid
  with fr columns always fits the content column, so it never clips or scrolls.
*/}

export const ModelLibrary = () => {
  const models = useModels();
  const order = [];
  const byTask = {};
  models.forEach((m) => {
    if (!byTask[m.taskDisplayName]) {
      byTask[m.taskDisplayName] = [];
      order.push(m.taskDisplayName);
    }
    byTask[m.taskDisplayName].push(m);
  });
  return (
    <div className="not-prose grid grid-cols-1 gap-4 md:grid-cols-2">
      {order.map((t) => (
        <a
          key={t}
          href={"/docs/" + taskSlug(t)}
          className="flex flex-col rounded-2xl border border-zinc-950/10 bg-white/40 p-5 no-underline transition-all hover:-translate-y-0.5 hover:border-zinc-950/30 hover:shadow-sm dark:border-white/10 dark:bg-white/[0.02] dark:hover:border-white/30"
        >
          <div className="flex items-center justify-between gap-2">
            <div className="font-semibold text-zinc-950 dark:text-white">{t}</div>
            <span className="shrink-0 rounded-full bg-zinc-950/[0.06] px-2 py-0.5 text-[11px] font-medium text-zinc-600 dark:bg-white/10 dark:text-zinc-300">
              {byTask[t].length} {byTask[t].length === 1 ? "model" : "models"}
            </span>
          </div>
          <div className="mt-3 flex flex-wrap items-center gap-2">
            {byTask[t].map((m) => (
              <ModelLogo key={m.modelId} src={pricingOf(m).favicon_url} alt={m.modelId} size={28} />
            ))}
          </div>
          <p className="mt-3 text-xs text-zinc-500 dark:text-zinc-400">
            {byTask[t].map((m) => m.modelId).join(", ")}
          </p>
        </a>
      ))}
    </div>
  );
};

## Using a model from the catalog in an API request

Once you've picked a model card, copy the **Model ID** and set it in your request:

```json
{
  "model": "zlm-v1-iab-classify-edge"
}
```

The rest of the request (headers and `input`) follows the standard `POST /v1/responses` format.
